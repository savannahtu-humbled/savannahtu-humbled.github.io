        function handleFile(file) {

            if (!file || !file.type.startsWith('image/')) return;

            const reader = new FileReader();

            reader.onload = e => {

                const img = new Image();

                img.onload = () => {

                    originalImage = img;

                    canvas.width = img.width;

                    canvas.height = img.height;

                   

                    // 初始化馬賽克層

                    mosaicLayer = document.createElement('canvas');

                    mosaicLayer.width = canvas.width;

                    mosaicLayer.height = canvas.height;

                   

                    initApp();

                    saveHistory();

                    render();

                };

                img.src = e.target.result;

            };

            reader.readAsDataURL(file);

        }

        // 3. 馬賽克畫筆邏輯

        function startDrawing(e) {

            if (!mosaicEnabled) return;

            isDrawing = true;

            draw(e);

        }

        function draw(e) {

            if (!isDrawing || !mosaicEnabled) return;

            const rect = canvas.getBoundingClientRect();

            const scaleX = canvas.width / rect.width;

            const scaleY = canvas.height / rect.height;

            const x = (e.clientX - rect.left) * scaleX;

            const y = (e.clientY - rect.top) * scaleY;

            const brushSize = parseInt(brushSlider.value);

            const pixelSize = parseInt(pixelSlider.value);

            // 在馬賽克層繪製

            const mCtx = mosaicLayer.getContext('2d');

           

            // 獲取該區域的顏色資訊 (基於目前 Canvas)

            const imageData = ctx.getImageData(x - brushSize/2, y - brushSize/2, brushSize, brushSize);

            const data = imageData.data;

            // 這裡我們直接在畫布上操作

            mCtx.save();

            mCtx.beginPath();

            mCtx.arc(x, y, brushSize / 2, 0, Math.PI * 2);

            mCtx.clip();

            // 繪製馬賽克方塊

            for (let my = y - brushSize/2; my < y + brushSize/2; my += pixelSize) {

                for (let mx = x - brushSize/2; mx < x + brushSize/2; mx += pixelSize) {

                    const sampleX = Math.floor(mx);

                    const sampleY = Math.floor(my);

                    const p = ctx.getImageData(sampleX, sampleY, 1, 1).data;

                    mCtx.fillStyle = `rgba(${p[0]},${p[1]},${p[2]},${p[3]/255})`;

                    mCtx.fillRect(mx, my, pixelSize, pixelSize);

                }

            }

            mCtx.restore();

            render(); // 重新組合渲染

        }

        function stopDrawing() {

                isDrawing = false;

                saveHistory();

            }

        }

        // 4. 渲染引擎

        function render() {

            if (!originalImage) return;

            // 清除畫布

            ctx.clearRect(0, 0, canvas.width, canvas.height);

           

            // A. CSS 濾鏡層

            let filterStr = `brightness(${sliders.brightness.el.value}%) blur(${sliders.blur.el.value}px)`;

            const f = filterSelect.value;

            if (f === 'grayscale') filterStr += ' grayscale(100%)';

            if (f === 'sepia') filterStr += ' sepia(100%)';

            if (f === 'invert') filterStr += ' invert(100%)';

           

            ctx.filter = filterStr;

            ctx.drawImage(originalImage, 0, 0);

            ctx.filter = 'none';

            // B. JS 像素處理層 (Dopamine, Y2K, Temp, Sharpen)

            const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);

            const data = imageData.data;

            if (f === 'dopamine') {

                for (let i = 0; i < data.length; i += 4) {

                    for (let j = 0; j < 3; j++) {

                        let c = data[i+j] / 255;

                        c = (c - 0.5) * 1.6 + 0.5;

                        data[i+j] = Math.min(255, Math.max(0, c * 255));

                    }

                    data[i] *= 1.1;

                }

            }

            if (f === 'y2k') {

                for (let i = 0; i < data.length; i += 4) {

                    data[i] = data[i] * 0.7 + 60;

                    data[i+1] = data[i+1] * 1.3;

                    data[i+2] = data[i+2] * 1.6;

                }

            }

            const temp = parseInt(sliders.temp.el.value);

            if (temp !== 0) {

                for (let i = 0; i < data.length; i += 4) {

                    data[i] += temp;

                    data[i+2] -= temp;

                }

            }

            const sharpen = parseFloat(sliders.sharpen.el.value);

            if (sharpen > 0) {

                const sharpened = applySharpen(imageData, sharpen);

                ctx.putImageData(sharpened, 0, 0);

            } else {

                ctx.putImageData(imageData, 0, 0);

            }

            // C. 馬賽克圖層疊加 (最上層)

            if (mosaicLayer) {

                ctx.drawImage(mosaicLayer, 0, 0);

            }

        }

        function applySharpen(imageData, intensity) {

            const w = imageData.width, h = imageData.height;

            const src = imageData.data;

            const output = ctx.createImageData(w, h);

            const dst = output.data;

            const a = -intensity, b = 1 + 4 * intensity;

            for (let y = 1; y < h - 1; y++) {

                for (let x = 1; x < w - 1; x++) {

                    const i = (y * w + x) * 4;

                    for (let c = 0; c < 3; c++) {

                        dst[i+c] = src[i+c-w*4]*a + src[i+c-4]*a + src[i+c]*b + src[i+c+4]*a + src[i+c+w*4]*a;

                    }

                    dst[i+3] = 255;

                }

            }

            return output;

        }
