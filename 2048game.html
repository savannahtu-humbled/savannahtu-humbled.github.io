    <script>

        const GRID_SIZE = 4;

        const CELL_SIZE = 67.5;

        const GAP = 10;

        let grid = [];

        let score = 0;

        let bestScore = localStorage.getItem('2048-best-score') || 0;

        let isGameOver = false;

        const tileContainer = document.getElementById('tile-container');

        const scoreDisplay = document.getElementById('score');

        const bestDisplay = document.getElementById('best-score');

        const gameOverDisplay = document.getElementById('game-over');

        const newGameBtn = document.getElementById('new-game');

        bestDisplay.textContent = bestScore;

        // 初始化遊戲

        function setupGame() {

            grid = Array(GRID_SIZE).fill().map(() => Array(GRID_SIZE).fill(0));

            score = 0;

            isGameOver = false;

            updateScore(0);

            gameOverDisplay.style.display = 'none';

            tileContainer.innerHTML = '';

           

            addRandomTile();

            addRandomTile();

            render();

        }

        // 新增隨機方塊 (2 或 4)

        function addRandomTile() {

            const emptyCells = [];

            for (let r = 0; r < GRID_SIZE; r++) {

                for (let c = 0; c < GRID_SIZE; c++) {

                    if (grid[r][c] === 0) emptyCells.push({r, c});

                }

            }

            if (emptyCells.length > 0) {

                const {r, c} = emptyCells[Math.floor(Math.random() * emptyCells.length)];

                grid[r][c] = Math.random() < 0.9 ? 2 : 4;

                createTileElement(r, c, grid[r][c], true);

            }

        }

        // 建立方塊 DOM

        function createTileElement(r, c, value, isNew = false) {

            const tile = document.createElement('div');

            tile.className = `tile tile-${value > 2048 ? 'super' : value} ${isNew ? 'tile-new' : ''}`;

            tile.textContent = value;

            updateTilePosition(tile, r, c);

            tile.id = `tile-${r}-${c}`;

            tileContainer.appendChild(tile);

            return tile;

        }

        function updateTilePosition(tile, r, c) {

            const x = c * (CELL_SIZE + GAP);

            const y = r * (CELL_SIZE + GAP);

            tile.style.transform = `translate(${x}px, ${y}px)`;

        }

        // 渲染畫面

        function render() {

            tileContainer.innerHTML = '';

            for (let r = 0; r < GRID_SIZE; r++) {

                for (let c = 0; c < GRID_SIZE; c++) {

                    if (grid[r][c] !== 0) {

                        createTileElement(r, c, grid[r][c]);

                    }

                }

            }

        }

        function updateScore(points) {

            score += points;

            scoreDisplay.textContent = score;

            if (score > bestScore) {

                bestScore = score;

                bestDisplay.textContent = bestScore;

                localStorage.setItem('2048-best-score', bestScore);

            }

        }

        // 移動邏輯

        function move(direction) {

            if (isGameOver) return;

            let moved = false;

            let mergedPoints = 0;

            // 旋轉與反轉邏輯以簡化四個方向的實作

            const rotatedGrid = getRotatedGrid(direction);

           

            for (let r = 0; r < GRID_SIZE; r++) {

                let row = rotatedGrid[r].filter(val => val !== 0);

                for (let i = 0; i < row.length - 1; i++) {

                    if (row[i] === row[i+1]) {

                        row[i] *= 2;

                        mergedPoints += row[i];

                        row.splice(i + 1, 1);

                        moved = true;

                    }

                }

                const newRow = row.concat(Array(GRID_SIZE - row.length).fill(0));

                if (JSON.stringify(rotatedGrid[r]) !== JSON.stringify(newRow)) {

                    moved = true;

                }

                rotatedGrid[r] = newRow;

            }

            if (moved) {

                grid = getRestoredGrid(rotatedGrid, direction);

                updateScore(mergedPoints);

                addRandomTile();

                render();

                checkGameOver();

            }

        }

        // 工具：根據方向旋轉矩陣

        function getRotatedGrid(direction) {

            let newGrid = JSON.parse(JSON.stringify(grid));

            if (direction === 'up') return rotate(rotate(rotate(newGrid)));

            if (direction === 'right') return rotate(rotate(newGrid));

            if (direction === 'down') return rotate(newGrid);

            return newGrid; // left is default

        }

        function getRestoredGrid(rotatedGrid, direction) {

            if (direction === 'up') return rotate(rotatedGrid);

            if (direction === 'right') return rotate(rotate(rotatedGrid));

            if (direction === 'down') return rotate(rotate(rotate(rotatedGrid)));

            return rotatedGrid;

        }

        function rotate(matrix) {

            const n = matrix.length;

            let res = Array(n).fill().map(() => Array(n).fill(0));

            for (let r = 0; r < n; r++) {

                for (let c = 0; c < n; c++) {

                    res[c][n - 1 - r] = matrix[r][c];

                }

            }

            return res;

        }

        function checkGameOver() {

            // 檢查是否有空位

            for (let r = 0; r < GRID_SIZE; r++) {

                for (let c = 0; c < GRID_SIZE; c++) {

                    if (grid[r][c] === 0) return;

                }

            }

            // 檢查是否有相鄰可合併的

            for (let r = 0; r < GRID_SIZE; r++) {

                for (let c = 0; c < GRID_SIZE; c++) {

                    if (c < GRID_SIZE - 1 && grid[r][c] === grid[r][c+1]) return;

                    if (r < GRID_SIZE - 1 && grid[r][c] === grid[r+1][c]) return;

                }

            }

           

            isGameOver = true;

            gameOverDisplay.style.display = 'flex';

        }

        // 事件監聽

        window.addEventListener('keydown', (e) => {

            if (['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight'].includes(e.key)) {

                e.preventDefault();

                const direction = e.key.replace('Arrow', '').toLowerCase();

                move(direction);

            }

        });

        newGameBtn.addEventListener('click', setupGame);

        // 觸控支援

        let touchStartX = 0;

        let touchStartY = 0;

        window.addEventListener('touchstart', (e) => {

            touchStartX = e.touches[0].clientX;

            touchStartY = e.touches[0].clientY;

        }, { passive: false });

        window.addEventListener('touchmove', (e) => {

            e.preventDefault();

        }, { passive: false });

        window.addEventListener('touchend', (e) => {

            const dx = e.changedTouches[0].clientX - touchStartX;

            const dy = e.changedTouches[0].clientY - touchStartY;

            const absDx = Math.abs(dx);

            const absDy = Math.abs(dy);

            if (Math.max(absDx, absDy) > 30) {

                if (absDx > absDy) {

                    move(dx > 0 ? 'right' : 'left');

                } else {

                    move(dy > 0 ? 'down' : 'up');

                }

            }

        });

        // 啟動

        window.onload = setupGame;

    </script>
