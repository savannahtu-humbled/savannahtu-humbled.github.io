        async function fetchYouBikeData() {

            setLoading(true);

            statusMsg.classList.add('hidden');

                filterAndRender();

            } catch (error) {

                console.error(error);

                showError('暫時無法取得即時資料，請檢查網路。');

            } finally {

            const areas = [...new Set(allData.map(item => item.sarea))].sort();

            areaSelect.innerHTML = '<option value="">所有區域</option>';

            areas.forEach(area => {

                areaSelect.appendChild(opt);

            });

        }

        function filterAndRender(customData = null) {

            if (allData.length === 0) return;

            const selectedArea = areaSelect.value;

            const keyword = searchInput.value.trim().toLowerCase();

            let displayData = customData || allData.filter(item => {

                const matchArea = selectedArea === "" || item.sarea === selectedArea;

                const matchKeyword = keyword === "" || item.ar.toLowerCase().includes(keyword) || item.sna.toLowerCase().includes(keyword);

                return matchArea && matchKeyword;

            });

            resultCount.textContent = `找到 ${displayData.length} 個站點`;

           

            renderMapMarkers(displayData);

            renderCards(displayData.slice(0, 30));

            if (displayData.length > 0 && !customData && (selectedArea || keyword)) {

                map.panTo([displayData[0].latitude, displayData[0].longitude]);

            }

        }

        // 尋找最近功能

        nearbyBtn.addEventListener('click', () => {

                showError('你的瀏覽器不支援定位功能。');

                return;

            }

            statusMsg.textContent = "正在獲取你的位置...";

            statusMsg.classList.remove('hidden', 'bg-red-100', 'text-red-700');

            statusMsg.classList.add('bg-blue-100', 'text-blue-700');

            navigator.geolocation.getCurrentPosition((position) => {

                const { latitude, longitude } = position.coords;

                if (userMarker) map.removeLayer(userMarker);

                userMarker = L.marker([latitude, longitude], {

                    icon: L.divIcon({

                        className: 'user-location-icon',

                    })

                }).addTo(map);

                const sorted = [...allData].map(station => {

                    const dist = getDistance(latitude, longitude, station.latitude, station.longitude);

                    return { ...station, distance: dist };

                }).sort((a, b) => a.distance - b.distance);

                const nearestFive = sorted.slice(0, 5);

                statusMsg.textContent = "已找到最近的 5 個站點！";

                map.setView([latitude, longitude], 15);

                filterAndRender(nearestFive);

            }, (error) => {

                showError('定位失敗，請確保已開啟權限。');

            });

        });

        function getDistance(lat1, lon1, lat2, lon2) {

            const R = 6371;

            const dLat = (lat2 - lat1) * Math.PI / 180;

            const dLon = (lon2 - lon1) * Math.PI / 180;

            const a = Math.sin(dLat/2) * Math.sin(dLat/2) +

                      Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2) * Math.sin(dLon/2);

            return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));

        }

        // 處理導航選點

        function handleRoutingPoint(station) {

            const cleanName = station.sna.replace('YouBike2.0_', '');

            if (!startPoint) {

                startPoint = station;

                statusMsg.textContent = `起點已選：${cleanName}。請點擊第二個站點作為終點。`;

            } else if (!endPoint && station.sno !== startPoint.sno) {

                endPoint = station;

                calculateRoute();

            }

        }

        function calculateRoute() {

            if (routingControl) map.removeControl(routingControl);

            routingControl = L.Routing.control({

                waypoints: [

                    L.latLng(startPoint.latitude, startPoint.longitude),

                    L.latLng(endPoint.latitude, endPoint.longitude)

                ],

                router: L.Routing.osrmv1({

                    serviceUrl: 'https://router.project-osrm.org/route/v1',

                    profile: 'bicycle' // 使用自行車路網

                }),

                lineOptions: {

                    styles: [{ color: '#10b981', weight: 6, opacity: 0.8 }]

                },

                addWaypoints: false,

                draggableWaypoints: false,

                createMarker: function() { return null; } // 不重複創建 Marker

            }).on('routesfound', function(e) {

                const routes = e.routes;

                const summary = routes[0].summary;

                const distKm = (summary.totalDistance / 1000).toFixed(2);

                const timeMin = Math.round(summary.totalTime / 60);

               

                routeInfo.innerHTML = `🏁 從 <b>${startPoint.sna.replace('YouBike2.0_', '')}</b> 到 <b>${endPoint.sna.replace('YouBike2.0_', '')}</b><br>

                                       距離：${distKm} 公里 | 預計騎乘：${timeMin} 分鐘`;

                routeInfo.classList.remove('hidden');

                statusMsg.classList.add('hidden');

                clearRouteBtn.classList.remove('hidden');

            }).addTo(map);

            isRoutingMode = false;

            routeBtn.classList.replace('bg-red-500', 'bg-emerald-600');

            routeBtnText.textContent = '🛣️ 導航';

        }

        function renderCards(stations) {

            dashboard.innerHTML = '';

            emptyState.classList.toggle('hidden', stations.length > 0);

            stations.forEach((station, index) => {

                const card = document.createElement('div');

                card.className = `bg-white p-4 rounded-2xl shadow-sm border border-slate-100 flex flex-col justify-between card-enter`;

                card.style.animationDelay = `${index * 0.03}s`;

                const cleanName = station.sna.replace('YouBike2.0_', '');

                const distText = station.distance ? `<span class="text-[10px] text-blue-500 font-bold ml-auto">約 ${station.distance.toFixed(2)} km</span>` : '';

                card.innerHTML = `

                    <div class="mb-3">

                        <div class="flex items-center gap-2 mb-1">
